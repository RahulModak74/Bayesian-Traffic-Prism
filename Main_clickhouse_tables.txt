 CREATE TABLE default.tracking10
(

    `timestamp` DateTime64(3, 'UTC'),
    `url` String,
    `referrer` String,
    `ip_address` String,
    `user_agent` String,
    `platform` String,
    `language` String,
    `hostname` String,
    `session_id` String,
    `browser_id` String,
    `fingerprint_id` String,
    `clickdata` String,
    `enhanced_data_id` Int64
)
ENGINE = MergeTree

SETTINGS index_granularity = 8192 │

CREATE TABLE default.live_sessions_mv
(
    `session_id` String,
    `browser_id` String,
    `last_activity` DateTime,
    `first_activity` DateTime,
    `url` String,
    `referrer` String,
    `ip_address` String,
    `user_agent` String,
    `hostname` String,
    `timestamp` DateTime64(3)
)
ENGINE = MergeTree
PARTITION BY toYYYYMM(last_activity)
ORDER BY (session_id, browser_id, timestamp)
TTL last_activity + toIntervalMinute(5)
SETTINGS index_granularity = 8192 │


CREATE TABLE default.users
(
    `username` String,
    `passwordHash` String,
    `hostname` String,
    `version` UInt32
)
ENGINE = ReplacingMergeTree(version)
ORDER BY (username, hostname)
SETTINGS index_granularity = 8192 │
